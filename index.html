import java.awt.*;
import java.awt.event.*;

public class _2_DesignCalciUsingGridLayoutAndHandleEvents extends Frame implements ActionListener{

    Font font = new Font("Times New Roman",Font.BOLD,18);
    String BtnStrArr[] = {"1","2","3","4","5","6","7","8","9","0","+","-","*","/","=","C"};
    TextField tf1 = new TextField(20);
    Button BtnArr[] = new Button [16];
    char CurrentOperator;
    int num1 = 0;
        int num2 = 0;
        int result = 0;

    _2_DesignCalciUsingGridLayoutAndHandleEvents(){
        setLayout(new BorderLayout());
        
        tf1.setFont(font);

        Panel p1 = new Panel();

        add(tf1,"North");
        add(p1,"Center");

        p1.setLayout(new GridLayout(4,4));
        for(int i = 0; i<16 ; i++)
        {
            BtnArr[i] = new Button(BtnStrArr[i]);
            BtnArr[i].addActionListener(this);
            p1.add(BtnArr[i]);
        }
    }

    public void actionPerformed(ActionEvent ae)
    {
        String EnteredNums = tf1.getText();

        for(int i=0; i<10; i++)
        {
            if (ae.getSource() == BtnArr[i]) {
                EnteredNums += BtnStrArr[i];
                tf1.setText(EnteredNums);
            }
        }
        if (ae.getActionCommand().equals("+")) {
            CurrentOperator = '+';
            num1 = Integer.parseInt(tf1.getText());
            tf1.setText("");
        }
        else if (ae.getActionCommand().equals("-")) {
            CurrentOperator = '-';
            num1 = Integer.parseInt(tf1.getText());
            tf1.setText("");
        }
        else if (ae.getActionCommand().equals("*")) {
            CurrentOperator = '*';
            num1 = Integer.parseInt(tf1.getText());
            tf1.setText("");
        }
        else if (ae.getActionCommand().equals("/")) {
            CurrentOperator = '/';
            num1 = Integer.parseInt(tf1.getText());
            tf1.setText("");
        }
        else if (ae.getActionCommand().equals("C")) {
            tf1.setText("");
            num1 = num2 = result = 0;
            CurrentOperator = '\0';
        }
        if (ae.getActionCommand().equals("=")) {
            num2 = Integer.parseInt(tf1.getText());
            switch (CurrentOperator) {
                case '+':
                        result = num1 + num2;
                        tf1.setText(""+(num1 + num2));
                        break;
                case '-':
                        result = num1 - num2;
                        tf1.setText(""+result);
                        break;
                case '*':
                        result = num1 * num2;
                        tf1.setText(""+result);
                        break;
                case '/':
                        result = num1 / num2;
                        tf1.setText(""+result);
                        break;
            
                default:
                        tf1.setText("Invalid Choice");
                        break;
            }
        }
    }

    public static void main(String[] args) {
        _2_DesignCalciUsingGridLayoutAndHandleEvents d1 = new _2_DesignCalciUsingGridLayoutAndHandleEvents();
        d1.setTitle("_2_DesignCalciUsingGridLayoutAndHandleEvents");
        d1.setSize(500,500);
        d1.setVisible(true);
    }
}

import java.net.*;
import java.io.*;

public class _8_TcpIpServer {
    public static void main(String[] args) throws Exception{
        ServerSocket ss = new ServerSocket(1111);
        System.out.println("Server is Running.....");
        Socket s1 = ss.accept();

        InputStream input = s1.getInputStream();
        OutputStream output = s1.getOutputStream();
        int c;
        while((c = input.read()) != -1)
        {
            System.out.print((char)c);
        }
        ss.close();
    }
}
